## let

将结果传递给表达式，会返回最后一行

fun letuse(){
    //第一个元素是1  将1传递吃传递给lambda
    listOf(1,2,3,434).first().let {
        println(it)

        it * it  //将最后一行返回
    }

    var name = ""
    name?.let { "" } ?: ""
//    ?: → 如果左边为 null，就使用右边的值。
}
